15    @staticmethod
16    def count_docx_pages(file_bytes: bytes) -> int:
17        """Extract the page count from a DOCX file.
18
19        Args:
20            file_bytes: Raw bytes of the DOCX file.
21
22        Returns:
23            int: Number of pages in the document. Returns 0 if unable to determine.
24        """
25        try:
26            with (
27                io.BytesIO(file_bytes) as bytes_io,
28                zipfile.ZipFile(bytes_io) as archive,
29            ):
30                try:
31                    app_xml = archive.read("docProps/app.xml").decode("utf-8")
32                    matches = re.findall(r"<Pages>(\d+)</Pages>", app_xml)
33                    return int(matches[0]) if matches else 0
34                except Exception:
35                    return 0
36        except Exception:
37            return 0
38
39    @staticmethod
40    def process_pdf(file_bytes: bytes) -> tuple[int, str]:
41        """Process PDF file and return page count and text content.
42
43        Args:
44            file_bytes: Raw bytes of the PDF file.
45
46        Returns:
47            Tuple containing:
48                - Number of pages in the document
49                - Extracted text content
50        """
51        reader = PdfReader(io.BytesIO(file_bytes))
52        pages = len(reader.pages)
53        text = " ".join(page.extract_text() for page in reader.pages)
54        return pages, text
55
56    @staticmethod
57    def process_docx(file_bytes: bytes) -> tuple[int, str]:
58        """Process DOCX file and return page count and text content.
59
60        Args:
61            file_bytes: Raw bytes of the DOCX file.
62
63        Returns:
64            Tuple containing:
65                - Number of pages in the document
66                - Extracted text content
67        """
68        pages = DocumentService.count_docx_pages(file_bytes)
69        doc = Document(io.BytesIO(file_bytes))
70
71        text_parts = [paragraph.text for paragraph in doc.paragraphs]
72        for table in doc.tables:
73            for row in table.rows:
74                for cell in row.cells:
75                    text_parts.append(cell.text)
76
77        return pages, " ".join(text_parts)

 33    def extract_content(self, file_bytes: bytes) -> tuple[int, str]:
 34        """Extract content from a resume file.
 35
 36        Args:
 37            file_bytes: Raw bytes of the resume file (PDF or DOCX)
 38
 39        Returns:
 40            Tuple containing page count and extracted text
 41
 42        Raises:
 43            ValueError: If the file format is not supported or content
 44                cannot be extracted
 45        """
 46        # Try PDF first
 47        try:
 48            return self.doc_service.process_pdf(file_bytes)
 49        except Exception:
 50            # If PDF fails, try DOCX
 51            try:
 52                pages, text = self.doc_service.process_docx(file_bytes)
 53                if pages == 0:
 54                    raise ValueError("Could not determine page count from DOCX file")
 55                return pages, text
 56            except Exception as e:
 57                raise ValueError("Invalid file format. Must be PDF or DOCX") from e
 58
 59    def validate_resume(self, file_bytes: bytes) -> ValidationResult:
 60        """Validate a resume file.
 61
 62        Performs the following checks:
 63        - File can be parsed as PDF or DOCX
 64        - Document contains sufficient content (>= 100 characters)
 65        - Document does not exceed maximum length (2 pages)
 66
 67        Args:
 68            file_bytes: Raw bytes of the resume file
 69
 70        Returns:
 71            ValidationResult containing the validation status and details
 72        """
 73        try:
 74            pages, text = self.extract_content(file_bytes)
 75        except ValueError as e:
 76            return ValidationResult(is_valid=False, error_message=str(e))
 77
 78        # Check minimum content length
 79        if len(text.strip()) < 100:
 80            return ValidationResult(
 81                is_valid=False,
 82                error_message="Resume must contain at least 100 characters",
 83                page_count=pages,
 84                text_content=text,
 85            )
 86
 87        # Check maximum page count
 88        if pages > 2:
 89            return ValidationResult(
 90                is_valid=False,
 91                error_message="Resume must not exceed 2 pages",
 92                page_count=pages,
 93                text_content=text,
 94            )
 95
 96        return ValidationResult(
 97            is_valid=True,
 98            page_count=pages,
 99            text_content=text,
100        )

 42    def fetch_talentfair_jobs(self) -> list[Job]:
 43        """Fetch jobs from Supabase and convert to Django models."""
 44        response = (  # type: ignore
 45            self.supabase.table("job_postings")  # type: ignore
 46            .select(
 47                "id, job_title, external_job_title, body, employer, location, "
 48                "source, apply_url, remote_status, min_pay, max_pay, pay_type"
 49            )
 50            .limit(10)
 51            .execute()
 52        )
 53
 54        jobs: list[Job] = []
 55        for job_data_raw in response.data:  # type: ignore
 56            job_data: JobSourceData = job_data_raw  # type: ignore
 57
 58            # Create or get company
 59            company, _ = Company.objects.get_or_create(
 60                id=secrets.token_hex(16),
 61                defaults={"company_name": job_data["employer"]},
 62            )
 63
 64            # Parse location data
 65            location_obj = None
 66            try:
 67                location_data = json.loads(job_data["location"])  # type: ignore
 68                location_obj, _ = Location.objects.get_or_create(**location_data)
 69            except (json.JSONDecodeError, TypeError):
 70                pass
 71
 72            # Create job instance
 73            job = Job(
 74                id=job_data["id"],
 75                source=job_data["source"],
 76                job_title=job_data["external_job_title"],
 77                description=job_data["body"],
 78                apply_url=job_data["apply_url"],
 79                company=company,
 80                location=location_obj,
 81                remote_status=job_data.get("remote_status", Job.RemoteStatus.ON_SITE),  # type: ignore
 82                min_pay=job_data.get("min_pay"),  # type: ignore
 83                max_pay=job_data.get("max_pay"),  # type: ignore
 84                pay_type=job_data.get("pay_type"),  # type: ignore
 85            )
 86            jobs.append(job)
 87
 88        return jobs
